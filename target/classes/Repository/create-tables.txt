DROP TABLE IF EXISTS Center
DROP TABLE IF EXISTS Blockchain
DROP TABLE IF EXISTS Missioni
DROP TABLE IF EXISTS Team
CREATE TABLE Missioni(ID_Missione VARCHAR(255) PRIMARY KEY, ID_Team VARCHAR(255), ID_Emergenza VARCHAR(255), Timestamp VARCHAR(255) )
CREATE TABLE Team(ID_Team VARCHAR(255) PRIMARY KEY, id_e1 VARCHAR(20), id_e2 VARCHAR(20), id_e3 VARCHAR(20))
CREATE TABLE Blockchain(Hash VARCHAR(255), PreviousHash VARCHAR(255), Timestamp VARCHAR(60), Data VARCHAR(5000))
CREATE TABLE Center(id MEDIUMINT NOT NULL AUTO_INCREMENT PRIMARY KEY, ID_mission VARCHAR(255) NOT NULL, Timestamp VARCHAR(255), Persona VARCHAR(100), Messaggio VARCHAR(255))
